SUPABASE SETUP REQUIRED

Step 1. Go to Supabase Dashboard › SQL Editor › New query › paste this:

create table if not exists beacons (
  beacon_id  text primary key,
  data       jsonb,
  updated_at timestamptz default now()
);
create table if not exists history (
  id         bigserial primary key,
  beacon_id  text,
  data       jsonb,
  created_at timestamptz default now()
);
alter table beacons enable row level security;
alter table history enable row level security;
create policy "anon_all_beacons" on beacons
  for all to anon using (true) with check (true);
create policy "anon_all_history" on history
  for all to anon using (true) with check (true);

Step 2. Click Run, then reload this page.
